exports.isAuthorized = function(req, res, next){
	
	checkToken(req, res, [1,2,3], function(err){
		if(err){
			return res.status(Boom.LOGIN_REQUIRED.statusCode).json(Boom.LOGIN_REQUIRED);

}; // End isAuthorized

function checkToken(req, res, types, callback){
	// check header or url parameters or post parameters for token
	var token = req.body.token || req.query.token || req.headers['x-access-token'];
	// decode token
	if (token) {
	// verifies secret and checks exp
		Session.findByToken(false, token, function(err,obj) {
			if (err) {
				callback(err);
			} else if (!obj.length){
				return res.status(Boom.LOGIN_REQUIRED.statusCode).json(Boom.LOGIN_REQUIRED);
			} else {
 
				let user = obj[0];
				if(user.message) {
					return res.status(401).json({status : 0, statusCode : 401, error : "Unauthorized", message : user.message });
				}
 
				let currTime = Math.round(+new Date()/1000);
				let expTime = user.exp_time;
 
				let validType = types.indexOf(user.meta.type) >= 0 ? 1 : 0;
 
				if(currTime > expTime){
					Session.deleteByToken(false, token, function(err){
						if(err){
							console.log(err);
						}
					});
					return res.status(Boom.LOGIN_REQUIRED.statusCode).json(Boom.LOGIN_REQUIRED);
				} else if (!validType){
					return res.status(Boom.NOT_ALLOWED.statusCode).json(Boom.NOT_ALLOWED);
				} else {
					req.user = user.meta;
					callback(null);
				}
			}
		});
  } else {
  	// return res.status(Boom.LOGIN_REQUIRED.statusCode).json(Boom.LOGIN_REQUIRED);
  	callback(' for isUserDataFound ');
  }
 
} // End checkToken
